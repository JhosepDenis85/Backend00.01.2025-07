let n3 = Number(inputN3.value);
let n4 = Number(inputN4.value);

    let resultado = "";

    // ¿Cuántos son pares?
    let pares = [n1, n2, n3, n4].filter(n => n % 2 === 0).length;
    resultado += `Cantidad de números pares: ${pares}\n`;

    // ¿Cuál es el mayor?
    let mayor = Math.max(n1, n2, n3, n4);
    resultado += `Mayor número: ${mayor}\n`;

    // Si el tercero es par, cuadrado del segundo

        resultado += `Cuadrado del segundo número: ${n2 ** 2}\n`;
    }

    // Si el primero < cuarto, calcular media
    if (n1 < n4) {
        let media = (n1 + n2 + n3 + n4) / 4;
        resultado += `Media de los 4 números: ${media.toFixed(2)}\n`;
    }

    // Si segundo > tercero y tercero entre 50 y 700, sumar todos
    if (n2 > n3 && n3 >= 50 && n3 <= 700) {
        let suma = n1 + n2 + n3 + n4;
        resultado += `Suma de los 4 números: ${suma}\n`;

}
